538  float arcLengthToTime (float s)
539  {
540  if (key_positions.size() < 2)
541  return -1.0;
542 
543  if (s < 0.0)
544  return 0;
545 
546  if (s > path_length)
547  return path_length;
548 
549  float arc_length = s;
550 
551  // find in which Bezir segment we should look
552  unsigned int segment = 0;
553  for (; segment < key_positions.size()-1; ++segment)
554  {
555  if (arc_lengths[segment+1][0] > arc_length)
556  break;
557  }
558 
559  // look inside Bezir segment in which linear approximation we should look for
560  unsigned int sub_seg = 0;
561  for (; sub_seg < arc_lengths[segment].size()-1; ++sub_seg)
562  {
563  if (arc_lengths[segment][sub_seg+1] > arc_length)
564  break;
565  }
566 
567  // convert from arc length (s) to time parameter (t)
568  // percentage inside sub segment
569  float alpha = (arc_length - arc_lengths[segment][sub_seg]) /
570  (arc_lengths[segment][sub_seg+1] - arc_lengths[segment][sub_seg]);
571 
572  // find parameter [0,1] inside Bezir segment
573  float t_ini = sub_seg / (float)(arc_lengths[segment].size()-1);
574  float t_end = (sub_seg+1) / (float)(arc_lengths[segment].size()-1);
575  float t_s = t_ini + alpha * (t_end - t_ini);
576 
577  // convert to global parameter of the path
578  t_s = t_s / (float)(key_positions.size()-1);
579 
580  // add time to beginning of segment
581  t_s += segment / (float)(key_positions.size()-1);
582 
583  return t_s;
584  }

727  void computeInnerControlPoints ( void )
728  {
729 
730  control_points_1.clear();
731  control_points_2.clear();
732 
733  // if there are only two keypoints it is a linear segment
734  // just distribute control points equally along segment
735  if (key_positions.size() == 2)
736  {
737  Eigen::Vector4f pt = 0.75*key_positions[0] + 0.25*key_positions[1];
738  control_points_1.push_back(pt);
739  control_points_1.push_back(pt);
740  pt << 0.25*key_positions[0] + 0.75*key_positions[1];
741  control_points_2.push_back(pt);
742  control_points_2.push_back(pt);
743  return;
744  }
745 
746  const int n = key_positions.size()-1;
747 
748  control_points_1.resize(n+1);
749  control_points_2.resize(n+1);
750 
751  // in theory this should be a nxn matrix, but since it is a tridiagonal matrix
752  // we are just storing the non-zero elements as a, b, and c
753  vector <float> a_orig;
754  vector <float> b_orig;
755  vector <float> c_orig;
756  vector <float> a;
757  vector <float> b;
758  vector <float> c;
759  vector <float> d;
760  a_orig.resize(n);
761  b_orig.resize(n);
762  c_orig.resize(n);
763  a.resize(n);
764  b.resize(n);
765  c.resize(n);
766  d.resize(n);
767 
768  // build the weight matrix, it is the same for all lines except first and last
769  a_orig[0] = 0.0;
770  b_orig[0] = 2.0;
771  c_orig[0] = 1.0;
772 
773  for (int i = 1; i < n-1; i++)
774  {
775  a_orig[i] = 1.0;
776  b_orig[i] = 4.0;
777  c_orig[i] = 1.0;
778  }
779 
780  a_orig[n-1] = 2.0;
781  b_orig[n-1] = 7.0;
782  c_orig[n-1] = 0.0;
783 
784  // solve using Thomas Algorithm for tridiagonal matrices
785  // solve the system three times, one for each coordinate (x, y, and z)
786  // we copy the coefficients each time to avoid recomputing them
787  for (int coord = 0; coord < 3; ++coord)
788  {
789  // build solution vector of linear system
790  d[0] = key_positions[0][coord] + 2.0 * key_positions[1][coord];
791  for (int i = 1; i < n-1; i++)
792  d[i] = 4.0*key_positions[i][coord] + 2.0*key_positions[i+1][coord];
793  d[n-1] = 8.0*key_positions[n-1][coord] + key_positions[n][coord];
794 
795  // copy coefficients, we are going to reuse them for every coordinate
796  a = a_orig;
797  b = b_orig;
798  c = c_orig;
799 
800  // forward sweep
801  c[0] = c[0]/b[0];
802  d[0] = d[0]/b[0];
803 
804  for (int i = 1; i < n; ++i)
805  {
806  float m = b[i] - a[i]*c[i-1];
807  c[i] = c[i] / m;
808  d[i] = (d[i] - a[i]*d[i-1]) / m;
809  }
810 
811  // back substitution
812  control_points_1[n-1][coord] = d[n-1];
813  for (int i = n-2; i >= 0; --i)
814  {
815  control_points_1[i][coord] = d[i] - c[i]*control_points_1[i+1][coord];
816  }
817  // add one last control point to facilitate future operations (same size as key vector)
818  control_points_1[n][coord] = control_points_1[n-1][coord];
819  }
820 
821  // set w = 1 for all control points
822  for (int i = 0; i <= n; ++i)
823  {
824  control_points_1[i][3] = 1.0;
825  }
826 
827  // control point 2 can be determined directly now
828  Eigen::Vector4f pt;
829  for (int i = 0; i < n-1; ++i)
830  {
831  pt = 2.0* key_positions[i+1] - control_points_1[i+1];
832  control_points_2[i] = pt;
833  }
834  pt = (key_positions[n] + control_points_1[n-1])*0.5;
835  control_points_2[n-1] = pt;
836  control_points_2[n] = pt;
837  }

844  void computeArcLength (void)
845  {
846  float divs = 100.0;
847  Eigen::Vector4f p0;
848  Eigen::Vector4f p1;
849  float dist = 0.0;
850  path_length = 0.0;
851 
852  arc_lengths.clear();
853 
854  p0 = key_positions[0];
855  for (unsigned int seg = 0; seg < key_positions.size()-1; ++seg)
856  {
857  vector <float> seg_lengths;
858  for (int i = 0; i < divs; ++i)
859  {
860  p1 = pointOnSegment( (float)i/divs, seg);
861  dist += (p1-p0).norm();
862  seg_lengths.push_back(dist);
863  p0 = p1;
864  }
865  // to make life easier when converting from t to s,
866  // we also repeat the last sub segment length at the end
867  // even though it will be covered by the first sub segment
868  // of the next Bezir segment
869  // note that we do not increment dist, since it will be done
870  // in the next iteration
871  p1 = key_positions[seg+1];
872  seg_lengths.push_back( dist + (p1-p0).norm() );
873 
874  arc_lengths.push_back (seg_lengths);
875  }
876  // the last segment connecting to the final key position
877  // there is no segment starting at this key position, so it
878  // must be treated apart
879  p1 = key_positions[key_positions.size()-1];
880  dist += (p1-p0).norm();
881 
882  path_length = dist;
883 
884  // insert a last vector with only the total distance
885  // makes life easier when converting from arc length to time parameter
